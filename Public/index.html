const express = require('express'); const path = require('path'); const app = express(); const port = 3000; // Serve static files from the "public" directory app.use(express.static(path.join(__dirname, 'public'))); // Optional: fallback route for root app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'indux.html')); // make sure filename matches }); // Listen on all interfaces app.listen(port, '0.0.0.0', () => { console.log(`App running on port ${port}`); });
